CREATE VIEW MEAL_ORDER_WITH_DETAILS AS 
    SELECT MO.*, M.PRICE, M.CALORIES FROM MEAL_ORDER MO, MEAL M 
    WHERE MO.MEAL_NAME = M.NAME;

SELECT MEAL_NAME, COUNT(*) AS QUANTI_PASTI 
FROM MEAL_ORDER 
GROUP BY MEAL_NAME;

SELECT SUM(PRICE) AS TOTALE_MAGG_15 
FROM MEAL WHERE PRICE > 15;

SELECT NAME, CALORIES, COUNT(*) AS CALORIE_SUP_500
FROM MEAL WHERE CALORIES > 500
GROUP BY CALORIES;

SELECT SUM(PRICE) AS SUM_PREZZI_CON_S 
FROM MEAL
WHERE NAME LIKE '%s%';

SELECT TIME_MEAL_ORDERED, COUNT(*) AS NUMERO_ORDINI
FROM MEAL_ORDER
WHERE TIME_MEAL_ORDERED < '2023-08-03 18:00:00'
GROUP BY MEAL_NAME;
